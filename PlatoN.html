def calc_porc   #Método que calcula los porcentajes nutricionales (% de proteínas, % de lípidos, etc.).
  cj_a, cj_c = @cj_alimento.head, @cj_cantidad.head
  while !cj_a.next.nil? #Recorremos las listas de alimentos y cantidades
    @porc_prote = @porc_prote + cj_a.value.proteinas * (cj_c.value / 1000) * 4  #Vamos calculando los acumulados los valores nutricionales, ya convertidos a kcal
    @porc_hidrat += cj_a.value.carbohidratos* (cj_c.value / 1000) * 4
    @porc_lipidos += cj_a.value.lipidos * (cj_c.value / 1000) * 9
    @vct = @vct + cj_a.value.total_kcal #Almacenamos el valor calórico total de los alimentos
    cj_a = cj_a.next
    cj_c = cj_c.next
  end
  @porc_prote = @porc_prote +  cj_a.value.proteinas * (cj_c.value / 1000) * 4
  @porc_hidrat += cj_a.value.carbohidratos* (cj_c.value / 1000) * 4
  @porc_lipidos += cj_a.value.lipidos * (cj_c.value / 1000) * 9
  @vct = @vct + cj_a.value.total_kcal

  @porc_prote = (porc_prote / @vct * 100).round(2)    #Finalmente, convertimos los valores nutricionales en proporciones.
  @porc_hidrat = (porc_hidrat / @vct * 100).round(2)
  @porc_lipidos = (porc_lipidos / @vct * 100).round(2)
end
