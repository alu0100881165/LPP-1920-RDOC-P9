def <=>(other)      #Operador Spaceship, comprueba que se este comparando con otro alimento, y decide si es <=> en función de las kcal totales.
  if(!other.is_a? Alimento_obj)
    return nil
  end
  self.total_kcal <=> other.total_kcal
end

def total_kcal      #Calcula las calorías totales del alimento.
