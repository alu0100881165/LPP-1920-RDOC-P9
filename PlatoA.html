def calc_porc_a #Método que calcula los porcentajes de impacto ambiental de los alimentos.
  cj_a, cj_c = @cj_alimento.head, @cj_cantidad.head
  while !cj_a.next.nil?   #Recorremos las listas de alimentos y cantidades
    @co2_total = @co2_total + cj_a.value.gei * (cj_c.value / 1000)    #Calculamos los valores de co2 y terreno totales.
    @terreno_total = @terreno_total + cj_a.value.terreno * (cj_c.value / 1000)
    cj_a = cj_a.next
    cj_c = cj_c.next
  end
  @co2_total = @co2_total + cj_a.value.gei * (cj_c.value / 1000)
  @terreno_total = @terreno_total + cj_a.value.terreno * (cj_c.value / 1000)
end
